def self.deflate(string)

  z = Zlib::Deflate.new(Zlib::BEST_COMPRESSION)

  dst = z.deflate(string, Zlib::FINISH)

  z.close

  dst

end

def self.inflate(string)

  zstream = Zlib::Inflate.new

  buf = zstream.inflate(string)

  zstream.finish

  zstream.close

  buf

end
